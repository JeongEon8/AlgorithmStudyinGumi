void bfs(int v){
    memset(visit, 0, sizeof(visit));
    queue<pair<int,int>> q;
    q.push({v, 0});
    visit[v] = 1;
    int friends = 0;

    while(!q.empty()){
        auto [now, d] = q.front(); q.pop();
        if(d > 1) continue;                 // κΉμ΄ 2κΉμ§€λ§

}
